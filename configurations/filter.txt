// test file to check parameter handler
// should give a simulator for bacteria in a box

set Simulator type = Bacteria
set Time step = 0.001 // :0.1:1   // looped
set Run time = 20.0
set Save period = 0.1
// set Output directory = test // given from command line

subsection Advection
	set Velocity type = Stokes
	set Maximum velocity = 2     			// doesn't matter if flow is off
	set Vortex radius = 0        			// also doesn't matter here
	set Vortex circulation = 0
	set Stokes refinement = 3 // note currently gives error if set to 0 and using stokes***
end

subsection Geometry
	set Geometry type = Filter 	// Box, File, or special type (see below)
	//set Geometry file = file.txt // file path relative to local directory
	// set Bottom left = {0,0}
	// set Top right = {10,5} // may be updated for filter or other ...

	// special geometry types: "Filter, Mixer, Splitter, File"
	subsection Filter
		set Number channels = 3
		set Channel thickness = 1.5
		set Wall thickness = 0.01
		set Left length = 5
		set Center length = 8
		set Right length = 1
	end

end // can also include obstacles ... need geometry file...

subsection Mesh
	// set Mesh type = Box // should be obtained from geometry
	set Global refinement = 1
	set Obstacle refinement = 2
	set Boundary refinement = 1
	//set Mesh file = file_name
end

subsection Chemicals
	set Number chemicals = 2
	set Implementation = {FE, FE}
	set Diffusion = {5,15}
	set Decay rate = {50,15}
	set Save chemicals = False
	set Time step factor = 1
	set Viscosity beta = 0.1
end

subsection Bacteria
	set Number bacteria = 100
	set Number groups = 1
	set Initial locations = {{}}  // should match dimension and number of groups
	set Diffusion = 0.1
	set Edge buffer = 0.2
	set Secretion rate = {100,100}
	set Mutation rate = 0.00000
	set Mutation strength = 100; 
	set Deterministic number mutate = 0
	set Deterministic mutate time = 0
	set Reintroducing = True
	set Reintroduction period = 2.0
	set Left subdomain length = 3
end

subsection Fitness
	set Fitness type = OR
	set Chemical fitness = {7.5,-8.0}
	set Secretion cost = 0.02
	set Chemical saturation = {0.01,0.1}
end
